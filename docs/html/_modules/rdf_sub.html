def calc_rdf(lines, N, nbin, ni, units):
    ''' A code to calulate g(r), the radial distribution function, from Sarkas output.
    Assumed that the xyz data is normalized by a_i, the ion sphere radius.

    Parameters
    ----------
    lines : array_like
        Lines of output data from Sarkas .out file

    '''
    ai = 1.0
    if(units == 0):
        ai = (3/(4*np.pi*ni))**(1./3.)
        print("ai = ", ai)

    # particles info
    L = (4.0*np.pi*N/3.0)**(1.0/3.0) #boxsize 
    boxsize = L
    hboxsize = L/2
    numPart = N

    # for histogram 
    maxbin = nbin # bin size
    dr = (L/2)/maxbin
#     print(L/2, dr)
    hist = [0]*(maxbin+1) # initialize list
    
    # particles position array
    px = np.zeros(numPart)
    py = np.zeros(numPart)
    pz = np.zeros(numPart)

    # reading particles
    for ipart in range(numPart):
        xyz = lines[ipart]
        x, y, z, vx, vy, vz, ax, ay, az = xyz.split()
        #x, y, z = xyz.split()
        px[ipart] = float(x)/ai
        py[ipart] = float(y)/ai
        pz[ipart] = float(z)/ai
        
#    print("Now, looping over particle pairs and set a distance histogram...")
    for i in range(numPart):
        xi = px[i]
        yi = py[i]
        zi = pz[i]
            
        # for j > i only. In fact, RDF should be considered for i != j. 
        # So 2 times smaller than the real RDF. It is adjusted at the end of the code.
        for j in range(i+1, numPart):   
            xj = px[j]
            yj = py[j]
            zj = pz[j]

            dx = xi - xj
            dy = yi - yj
            dz = zi - zj

            # periodic condition applied for minimum image
            if(dx < -hboxsize):
                dx = dx + L
            if(dx > hboxsize):
                dx = dx - L
            if(dy < -hboxsize):
                dy = dy + L
            if(dy > hboxsize):
                dy = dy - L
            if(dz < -hboxsize):
                dz = dz + L
            if(dz > hboxsize):
                dz = dz - L

            r_sq = dx**2 + dy**2 + dz**2 
            r = np.sqrt(r_sq)
            ibin = int(np.ceil(r/dr))
            if (ibin <= maxbin):
                hist[ibin] += 1

# Normalization
    r = np.zeros(maxbin)
    rdf = np.zeros(maxbin)
    for i in range(1, maxbin+1):
        r[i-1] = (i - 0.5)*dr
        shell_vol = (i*dr)**3 - ((i-1)*dr)**3 
        rdf[i-1] = hist[i]/numPart/shell_vol*2 # 2 comes from i>j pairs. Need to convert to i!=j pairs.

    return r, rdf
